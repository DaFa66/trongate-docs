submit_login()

function submit_login(): void

Description

Handles the submission of login forms, validates user input, and logs users in if validation passes. Redirects to the login form on validation failure or redirects to the base URL on 'Cancel' submission.

Parameters

No parameters.

Return Value

void

Example Usage

function submit_login() {

    $submit = post('submit');

    if ($submit == 'Submit') {
        $this->validation_helper->set_rules('username', 'username', 'required|callback_login_check');
        $this->validation_helper->set_rules('password', 'password', 'required|min_length[5]');
        $result = $this->validation_helper->run();

        if ($result == true) {
            // Success!  Now log the user in...
        } else {
            // Form validation error - present the login form again.
            $this->login();
        }
    } elseif ($submit == 'Cancel') {
        // Send user back to homepage.
        redirect(BASE_URL);
    }
}

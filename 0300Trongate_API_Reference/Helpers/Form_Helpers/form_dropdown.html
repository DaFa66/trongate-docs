form_dropdown()

function form_dropdown(string $name, array $options, string|null $selected_key = null, array|null $attributes = null, string|null $additional_code = null): string

Description

Generates an HTML select menu.

Parameters

Parameter Type Description
$name string The name attribute for the select element.
$options array An associative array of options (value => text).
$selected_key string|null The key of the selected option, if any.
$attributes array|null An array of HTML attributes for the select element.
$additional_code string|null Additional HTML code to include in the select element.

Return Value

Type Description
string The generated HTML for the select menu.

Example Usage

  $name = 'country';
  $options = [
      'us' => 'United States',
      'uk' => 'United Kingdom',
      'ca' => 'Canada',
      'au' => 'Australia'
  ];
  $selected_key = 'uk';
  $attributes = ['class' => 'form-control'];
  $additional_code = 'id="countrySelect"';
  echo form_dropdown($name, $options, $selected_key, $attributes, $additional_code);
  // Output: HTML for a select menu with the United Kingdom option selected and additional attributes

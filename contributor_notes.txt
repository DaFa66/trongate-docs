    /**
     * Pre-token validation tests.
     *
     * This method performs validation tests before issuing a new token. In a development environment (ENV = 'dev'), it checks if 'user_id' is set and if it's a numeric value.
     *
     * @param array $input An array containing input data.
     * @return array The input data after validation checks.
     */
    function _pre_token_validation(array $input): array {
        // Validation tests that happen before a new token is issued

        if (ENV !== 'dev') {
            // Add your own validation code here!
            echo 'Pre-token validation failed: Forbidden';
            http_response_code(403);
            die();
        }

        if (!isset($input['user_id'])) {
            http_response_code(400);
            echo 'Pre-token validation failed: No user_id submitted!';
            die();
        } elseif (!is_numeric($input['user_id'])) {
            http_response_code(400);
            echo 'Pre-token validation failed: Non-numeric user_id submitted!';
            die();
        }

        return $input;
    }
